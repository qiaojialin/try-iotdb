set storage group to root.turbine
create timeseries root.turbine.d1.s1(temperature1) with datatype=FLOAT, encoding=GORILLA, compression=SNAPPY tags(unit=degree, owner=user1)
insert into root.turbine.d1(timestamp,s1,s1) values(1,1,2)
show timeseries root
count timeseries root
select * from root
select last * from root
select count(*) from root
select count(*) from root group by ([0,10), 2ms)
select last_value(*) from root group by ([0,10), 2ms) fill(float[previous])
